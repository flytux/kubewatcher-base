Namespace -> container Resource Limit : CPU Limit 설정 필요

apiVersion: monitoring.coreos.com/v1
kind: ServiceMonitor
metadata:
  name: actuator-metrics
  namespace: dev
spec:
  endpoints:
  - interval: 30s
    path: /actuator/prometheus
    port: web
    relabelings:
    - sourceLabels:
      - __meta_kubernetes_endpoints_label_serviceMonitor_type_01
      targetLabel: job
  selector:
    matchLabels:
      serviceMonitor-type-01: actuator-metrics

apiVersion: monitoring.coreos.com/v1
kind: ServiceMonitor
metadata:
  name: exporter-metrics
  namespace: dev
spec:
  endpoints:
  - interval: 30s
    path: /metrics
    port: exporter
    relabelings:
    - sourceLabels:
      - __meta_kubernetes_endpoints_label_serviceMonitor_type_02
      targetLabel: job
  selector:
